1번테이블 4중조인
select d.day_id, n.scenario_name, s.scene_num, f.*
from scene s
join film_plan f  on s.scene_id = f.scene_id
join scenario n on s.scenario_id = n.scenario_id
join schedule_day d on d.day_id = f.day_id
